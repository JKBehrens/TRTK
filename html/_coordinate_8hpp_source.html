01222 template <class T>
01223 Coordinate<T> Coordinate<T>::orthogonal() const
01224 {
01225     assert(this->size() >= 2);
01226     assert(!Tools::isZero(this->norm()));
01227 
01228 
01229     // Find largest and second largest element of this coordinate
01230     // and their according positions.
01231 
01232     T max[2] = {0};
01233     int pos[2] = {0};
01234 
01235     for (unsigned int i = 0; i < this->size(); ++i)
01236     {
01237         using std::abs;
01238 
01239         if (abs(m_data(i)) > abs(max[0]))
01240         {
01241             max[1] = max[0];
01242             max[0] = m_data(i);
01243             pos[1] = pos[0];
01244             pos[0] = i;
01245         }
01246         else if (abs(m_data(i)) >= abs(max[1]))
01247         {
01248             max[1] = m_data(i);
01249             pos[1] = i;
01250         }
01251     }
01252 
01253     // Construct the orthogonal vector by switching the positions of
01254     // the two largest elements and negating the second largest element.
01255     // All other components are set to zero.
01256     // The dot product of this constructed vector with *this always equals zero.
01257 
01258     Coordinate<T> result;
01259     result.resize(this->size(), 0);
01260     result[pos[0]] = - max[1];
01261     result[pos[1]] =   max[0];
01262 
01263     return result;
01264 }

01418 {

01662 std::istream & operator >> (std::istream & input, Coordinate<T> & coordinate)
01663 {
01664     // The input format is a comma separated list enclosed by round brackets.
01665     // Anything diverging from it, leads to an exception.
01666     //
01667     // Example: (1, 2, 3)
01668 
01669     char input_character;
01670     const std::string error_msg("Wrong input format or coordinate size.");
01671 
01672     // throw an exception if the input character differs from what was expected
01673 
01674     std::ios_base::iostate old_iostate = input.exceptions();
01675 
01676     input.exceptions(std::ios_base::badbit | std::ios_base::failbit);
01677 
01678     // read in coordinate
01679 
01680     input >> input_character;
01681     if (input_character != '(') throw std::invalid_argument(error_msg);
01682 
01683     int i = 0;
01684 
01685     while (i < signed(coordinate.size()) - 1)
01686     {
01687         input >> coordinate[i++];
01688 
01689         input >> input_character;
01690         if (input_character != ',') throw std::invalid_argument(error_msg);
01691     }
01692 
01693     if (coordinate.size() != 0) // allow empty coordinates like '( )'
01694     {
01695         input >> coordinate[i];
01696     }
01697 
01698     input >> input_character;
01699     if (input_character != ')') throw std::invalid_argument(error_msg);
01700 
01701     // restore old io_state
01702 
01703     input.exceptions(old_iostate);
01704 
01705     return input;
