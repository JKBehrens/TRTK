00154 std::pair<Eigen::Matrix<T, 3, 1>, T> axisAngleFromRotationMatrix(const Eigen::Matrix<T, 3, 3> & matrix)
00155 {
00156     // The implementation is based on the lecture notes of Carlo Tomasi. See [1] for more information.
00157     // [1] https://www2.cs.duke.edu/courses/fall13/compsci527/notes/rodrigues.pdf
00158 
00159     using namespace Eigen;
00160     using Matrix3T = Matrix<T, 3, 3>;
00161     using Vector3T = Matrix<T, 3, 1>;
00162 
00163     const T eps = 1000 * std::numeric_limits<T>::epsilon();
00164     const T pi = 3.14159265359;
00165 
00166     Matrix3T A = (matrix - matrix.transpose()) / 2;
00167     Vector3T rho = Vector3T(A(2, 1), A(0, 2), A(1, 0));
00168     T s = rho.norm();
00169     T c = (matrix.trace() - 1) / 2;
00170 
00171     Vector3T axis;
00172     T angle;
00173 
00174     if (abs(s - 0) < eps && abs(c - 1) < eps)
00175     {
00176         axis = Vector3T(1, 0, 0);
00177         angle = 0;
00178     }
00179     else if (abs(s - 0) < eps && abs(c + 1) < eps)
00180     {
00181         angle = pi;
00182         Matrix3T B = matrix + Matrix3T::Identity();
00183         for (int i = 0; i < 3; ++i)
00184         {
00185             axis = B.col(i).normalized();
00186             if (axis.norm() > 0) break;
00187         }
00188         // Assure uniqueness by forcing the axis onto the half-hemisphere.
00189         if (abs(axis.x()) < eps && abs(axis.y()) < eps && axis.z() < 0 ||
00190             abs(axis.x()) < eps && axis.y() < 0 ||
00191             axis.x() < 0)
00192         {
00193             angle = -angle;
00194             axis = -axis;
00195         }
00196     }
00197     else
00198     {
00199         using ::atan2;
00200         axis = rho / s;
00201         angle = atan2(s, c);
00202     }
00203 
00204     return std::make_pair(axis, angle);
00205 }

00678 TRTK::Coordinate<T> orthogonalMatrixToQuaternion(const Eigen::Matrix<T, 3, 3> & matrix)
00679 {
00680     // References:
00681     //
00682     // [1] http://www.euclideanspace.com/maths/geometry/rotations/conversions/matrixToQuaternion/index.htm
00683     // [2] http://en.wikipedia.org/wiki/Rotation_matrix
00684 
00685     // Note: Coordinate<T> is of the form (x, y, z, w) whereas quaternions are often notated (w, x, y, z)!
00686 
00687     using std::sqrt;
00688 
00689     TRTK::Coordinate<T> quaternion(0, 0, 0, 0);
00690 
00691     T trace = matrix.trace();
00692 
00693     if (trace > -1) // check because of the computation of the root
00694     {
00695         double s = 0.5 / sqrt(trace + 1);
00696         quaternion.x() = 0.25 / s;
00697         quaternion.y() = (matrix(2,1) - matrix(1,2)) * s;
00698         quaternion.z() = (matrix(0,2) - matrix(2,0)) * s;
00699         quaternion.w() = (matrix(1,0) - matrix(0,1)) * s;
00700     }
00701     else if (matrix(0,0) > matrix(1,1) && matrix(0,0) > matrix(2,2))
00702     {
00703         double s = 2.0 * sqrt(1 + matrix(0,0) - matrix(1,1) - matrix(2,2));
00704         quaternion.x() = (matrix(2,1) - matrix(1,2)) / s;
00705         quaternion.y() = 0.25 * s;
00706         quaternion.z() = (matrix(0,1) + matrix(1,0)) / s;
00707         quaternion.w() = (matrix(0,2) + matrix(2,0)) / s;
00708     }
00709     else if (matrix(1,1) > matrix(2,2))
00710     {
00711         double s = 2 * sqrt(1 + matrix(1,1) - matrix(0,0) - matrix(2,2));
00712         quaternion.x() = (matrix(0,2) - matrix(2,0)) / s;
00713         quaternion.y() = (matrix(0,1) + matrix(1,0)) / s;
00714         quaternion.z() = 0.25 * s;
00715         quaternion.w() = (matrix(1,2) + matrix(2,1)) / s;
00716     }
00717     else
00718     {
00719         double s = 2 * sqrt(1 + matrix(2,2) - matrix(0,0) - matrix(1,1));
00720         quaternion.x() = (matrix(1,0) - matrix(0,1)) / s;
00721         quaternion.y() = (matrix(0,2) + matrix(2,0)) / s;
00722         quaternion.z() = (matrix(1,2) + matrix(2,1)) / s;
00723         quaternion.w() = 0.25 * s;
00724     }
00725 
00726     return quaternion.normalize();
00727 }

00751 template <class T>
00752 TRTK::Coordinate<T> orthogonalMatrixToQuaternion2(const Eigen::Matrix<T, 3, 3> & matrix)
00753 {
00754     // See [1] for more details about the implementation.
00755 
00756     // The matrix K3 is selfadjoint because it is real and symmetric. Thus we can use an
00757     // appropriate solver. The solver employed below only uses the lower triangular part
00758     // of the matrix hence we can leave the upper part uninitialized.
00759 
00760     typedef Eigen::Matrix<T, 4, 4> Matrix4T;
00761     typedef Eigen::Matrix<T, 4, 1> Vector4T;
00762 
00763     Matrix4T K3;
00764 
00765     K3(0,0) = matrix(0,0) - matrix(1,1) - matrix(2,2);
00766     K3(1,0) = matrix(1,0) + matrix(0,1);
00767     K3(2,0) = matrix(2,0) + matrix(0,2);
00768     K3(3,0) = matrix(1,2) - matrix(2,1);
00769     K3(1,1) = matrix(1,1) - matrix(0,0) - matrix(2,2);
00770     K3(2,1) = matrix(2,1) + matrix(1,2);
00771     K3(3,1) = matrix(2,0) - matrix(0,2);
00772     K3(2,2) = matrix(2,2) - matrix(0,0) - matrix(1,1);
00773     K3(3,2) = matrix(0,1) - matrix(1,0);
00774     K3(3,3) = matrix(0,0) + matrix(1,1) + matrix(2,2);
00775 
00776     K3 /= 3;
00777 
00778     Eigen::SelfAdjointEigenSolver<Matrix4T> solver(K3);
00779 
00780     Vector4T quaternion = solver.eigenvectors().col(3).normalized();
00781 
00782     return Coordinate<T>(quaternion);
00783 }

00796 template <class T>
00797 Eigen::Matrix<T, 3, 3> quaternionToOrthogonalMatrix(T q0, T q1, T q2, T q3)
00798 {
00799     // Revised code.
00800 
00801     // Identical to what you can find in the Wikipedia article
00802     // https://en.wikipedia.org/wiki/Quaternions_and_spatial_rotation
00803     // This matrix is a left-handed (post-multiplied) rotation matrix.
00804 
00805     assert(abs(q0 * q0 + q1 * q1 + q2 * q2 + q3 * q3 - 1) < 1e-5); // check for unit quaternion
00806 
00807     Eigen::Matrix<T, 3, 3> R;
00808 
00809     R(0, 0) = 1 - 2 * (q2 * q2 + q3 * q3);
00810     R(0, 1) = 2 * q1 * q2 - 2 * q0 * q3;
00811     R(0, 2) = 2 * q1 * q3 + 2 * q0 * q2;
00812 
00813     R(1, 0) = 2 * q1 * q2 + 2 * q0 * q3;
00814     R(1, 1) = 1 - 2 * (q1 * q1  + q3 * q3);
00815     R(1, 2) = 2 * q2 * q3 - 2 * q0 * q1;
00816 
00817     R(2, 0) = 2 * q1 * q3 - 2 * q0 * q2;
00818     R(2, 1) = 2 * q2 * q3 + 2 * q0 * q1;
00819     R(2, 2) = 1 - 2 * (q1 * q1 + q2 * q2);
00820 
00821     return R;
00822 }

00936 template <class T>
00937 T randn()
00938 {
00939     /*
00940     This algorithm uses the Marsaglia polar method. For more details, please
00941     have a look at "A convenient method for generating normal variables" by
00942     G. Marsaglia and T. A. Bray (SIAM Rev. 6, 260-264, 1964).
00943     */
00944 
00945     // The polar method yields two random variables per computation. So one
00946     // random variable is cached each time and returned on the next call.
00947 
00948     using std::log;
00949     using std::sqrt;
00950 
00951     static bool cache_is_valid = false;
00952     static T cached_random_variable = 0;
00953 
00954     if (!cache_is_valid)
00955     {
00956         T v1, v2, s;
00957 
00958         do
00959         {
00960             v1 = rand<T>(-1, 1);
00961             v2 = rand<T>(-1, 1);
00962 
00963             s = v1 * v1 + v2 * v2;
00964         }
00965         while (s >= 1 || s == 0);
00966 
00967         const T t = sqrt(-2 * log(s) / s);
00968 
00969         T random_variable = v1 * t;
00970 
00971         cached_random_variable = v2 * t;
00972         cache_is_valid = true;
00973 
00974         return random_variable;
00975     }
00976     else
00977     {
00978         cache_is_valid = false;
00979 
00980         return cached_random_variable;
00981     }
00982 }

01061 template <class T>
01062 Eigen::Matrix<T, 3, 3> rotationMatrix(const Eigen::Matrix<T, 3, 1> & axis_, double angle)
01063 {
01064     using Vector3T = Eigen::Matrix<T, 3, 1>;
01065     using Matrix3T = Eigen::Matrix<T, 3, 3>;
01066 
01067     T axis_magnitude = axis_.norm();
01068     assert(axis_magnitude > 0); // Axis may not be a null vector.
01069     Vector3T axis = axis_ / axis_magnitude;
01070 
01071     // Find an orthogonal vector.
01072 
01073     const T eps = std::numeric_limits<T>::epsilon();
01074 
01075     auto isNullvector = [&] (Vector3T & vec) -> bool
01076     {
01077         return vec.norm() < 10 * eps;
01078     };
01079 
01080     auto isCollinear = [&] (Vector3T a, Vector3T b) -> bool
01081     {
01082         a.normalize();
01083         b.normalize();
01084         return abs(abs(a.dot(b)) - 1) < 10 * eps;
01085     };
01086 
01087     auto vec = Vector3T::Random().normalized();
01088     while (isNullvector(vec) || isCollinear(vec, axis))
01089     {
01090         vec = Vector3T::Random().normalized();
01091     }
01092 
01093     auto vec_orthonormal = (vec - vec.dot(axis) * axis).normalized(); // Cf. Gram-Schmidt process
01094     assert(abs(axis.dot(vec_orthonormal)) < 1000 * eps); // Axis and vec_orthonormal must be orthonormal. Implementation error.
01095 
01096     // Compute the rotation matrix
01097 
01098     // Make a change of basis such that the rotation axis is aligned
01099     // with the x-axis, rotate around this axis and revert the change of
01100     // basis.
01101 
01102     Matrix3T M;
01103     M.row(0) = axis;
01104     M.row(1) = vec_orthonormal;
01105     M.row(2) = axis.cross(vec_orthonormal);
01106 
01107     Matrix3T R_x;
01108     R_x << 1,          0,           0,
01109            0, cos(angle), -sin(angle),
01110            0, sin(angle),  cos(angle);
01111 
01112     return M.transpose() * R_x * M;
01113 }
